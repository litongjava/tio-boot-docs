删除多余的数据

```sql
SELECT *
FROM max_kb_document d
WHERE NOT EXISTS (
    SELECT 1
    FROM max_kb_dataset ds
    WHERE ds.id = d.dataset_id
);


DELETE FROM max_kb_document d
WHERE NOT EXISTS (
    SELECT 1
    FROM max_kb_dataset ds
    WHERE ds.id = d.dataset_id
);
```
